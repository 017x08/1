const TelegramBot = require('node-telegram-bot-api'); const admin = require('firebase-admin'); // আপনার টোকেন এবং চ্যাট আইডি const token = '8782276212:AAGRKZdMcmwhydJVVO-uN78bXZMEcV4zlno'; const myChatId = '8540900103'; // Firebase Setup // (firebase-key.json ফাইলটি আপনার ফায়ারবেস কনসোল থেকে ডাউনলোড করে নিবেন) const serviceAccount = require("./firebase-key.json"); admin.initializeApp({ credential: admin.credential.cert(serviceAccount), databaseURL: "YOUR_FIREBASE_DATABASE_URL" }); const bot = new TelegramBot(token, {polling: true}); const db = admin.database(); // ফাইল হ্যান্ডেল করার ফাংশন async function handleFile(msg, type) { if (msg.chat.id.toString() !== myChatId) return; // শুধুমাত্র আপনি পাঠালেই সেভ হবে let fileId; if (type === 'photo') fileId = msg.photo[msg.photo.length - 1].file_id; else if (type === 'video') fileId = msg.video.file_id; else fileId = msg.document.file_id; const fileLink = await bot.getFileLink(fileId); db.ref(type).push({ url: fileLink, name: msg.caption || type + "_" + Date.now(), timestamp: Date.now() }); bot.sendMessage(myChatId, `✅ ${type.toUpperCase()} আপনার ওয়েবসাইটে যোগ করা হয়েছে!`); } bot.on('photo', (msg) => handleFile(msg, 'photo')); bot.on('video', (msg) => handleFile(msg, 'video')); bot.on('document', (msg) => handleFile(msg, 'file'));
