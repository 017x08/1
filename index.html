const TelegramBot = require('node-telegram-bot-api'); const admin = require('firebase-admin'); // আপনার দেওয়া আইডি const token = '8782276212:AAGRKZdMcmwhydJVVO-uN78bXZMEcV4zlno'; const myChatId = '8782276212'; // আপনার আপডেট করা আইডি // Firebase Setup const serviceAccount = require("./firebase-key.json"); admin.initializeApp({ credential: admin.credential.cert(serviceAccount), databaseURL: "YOUR_FIREBASE_DATABASE_URL" // আপনার ফায়ারবেস ইউআরএল এখানে দিন }); const bot = new TelegramBot(token, {polling: true}); const db = admin.database(); async function handleFile(msg, type) { // আইডি চেক (নিরাপত্তার জন্য) if (msg.chat.id.toString() !== myChatId) return; let fileId; if (type === 'photo') fileId = msg.photo[msg.photo.length - 1].file_id; else if (type === 'video') fileId = msg.video.file_id; else fileId = msg.document.file_id; try { const fileLink = await bot.getFileLink(fileId); // Firebase এ ডাটা পাঠানো await db.ref(type).push({ url: fileLink, name: msg.caption || type + "_" + Date.now(), timestamp: Date.now() }); bot.sendMessage(myChatId, `✅ সফলভাবে আপনার ওয়েবসাইটের ${type} সেকশনে যোগ করা হয়েছে!`); } catch (error) { bot.sendMessage(myChatId, "❌ ফাইল সেভ করতে সমস্যা হয়েছে।"); } } bot.on('photo', (msg) => handleFile(msg, 'photo')); bot.on('video', (msg) => handleFile(msg, 'video')); bot.on('document', (msg) => handleFile(msg, 'file')); console.log("Bot is running...");
